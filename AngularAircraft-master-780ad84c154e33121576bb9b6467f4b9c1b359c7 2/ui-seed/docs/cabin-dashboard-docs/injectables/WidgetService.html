import { Injectable } from '@angular/core';
import { AngularFireDatabase, FirebaseListObservable, FirebaseObjectObservable } from 'angularfire2/database';
import { Observable } from 'rxjs/Observable';
import { Store } from '@ngrx/store';
import { getUserId } from '../../core/store/reducers/user.reducer'
import { Widget,SBBWidget } from '../model/widget'
import { Usage } from '../model/usage'
import { AppStore } from '../../core/store/app-store'
import { NGXLogger } from 'ngx-logger';

import * as _ from 'lodash';

@Injectable()
export class WidgetService {

  private widgets:  FirebaseListObservable<Widget[]>;
  userId;
  
  /**
   * Represents our Widget Service
   * @constructor
   * @param {AngularFireDatabase} db - connection to our db
   * @param {UserService} user - for managing user  widgets
   */
  constructor(private db: AngularFireDatabase,
	      private store: Store<AppStore>,
	      private logger: NGXLogger
	     ) {
    // the database will eventually move to store so we just pass the parameter string of the stuff we want.
    this.store.select(getUserId).combineLatest().subscribe(id => this.userId = id);
  }

  /**
   * types fetches widget types
   * @return {FirebaseListObservable} Observable for firebase
   */
  types(): FirebaseListObservable<any[]>{
    return this.db.list('/widget_types/');
  }
  
  /**
  * fetch - get the user widgets
  *
  */
  fetch(): Observable<Widget[]>{
    this.widgets =  this.db.list(`/dashboard_user/${this.userId}/widgets`);
    this.logger.debug("fetching widgets");
    // This is where the magic happens
    return this.widgets.switchMap((widgets:Widget[]) => {
	// change name to widget data
      // Get the payload for the widget in this case we only have plans
      let widgetData$ = this.widgetData(widgets);
      // Now we must combine the data

      return Observable.combineLatest(widgetData$, (...widData) => {
	return widgets.map((widget, index) => {
	  const data = widData[index];
	  widget.id = widget.$key;
	  if (widget.data !== data) {
	    this.logger.warn("No data for widget " + widget.id);
            return Object.assign({}, widget, { data });
	  }
          return widget;
	})});
    })
  }

  /**
   * multipleUsages - fetches multiple plans usage
   * @param {array} planIDs : array of planIds to fetch
   * @param {Object} query : specify query for historical data.
   * @return {Observable<any> } : either a list of widget data or singleton
   */  
  multipleUsages(planIDs:any[]):Observable<Usage[]>{  
    let planD = planIDs.map(id => this.usageData(id));
    return Observable.combineLatest(planD, (...planD) => {
      let d$ = []
      planD.forEach(p => d$.push(p as Usage));		    
      return d$;
    });  
  }
  
  /**
   * historicalUsage - Gets historical data for history widgets
   * @param {string} planID : planid to fetch
   * @param {Object} query : specify query for historical data.
   * @return {Observable<any> } : either a list of widget data or singleton
   */  
  historicalUsage(planID,query = {}):Observable<any>{
    if(!planID){
      return Observable.of({});
    }
//we need to refactor the db setup
    return this.db.list(`/usage/${planID}/historical`, {query:{orderByKey:true}}).switchMap((d) => {
      let data = [];
      return Observable.of(data);
    });
    
    //   .catch(() => {
    // 	this.logger.warn("No data discovered for planID" + planID);
    // 	return Observable.of({});
    // });    
  }

  /**
   * usageData - retrieves data for plan
   * @param {string} planID : planid to fetch
   * @param {Object} query : specify query for historical data.
   * @return {Observable<any> } : either a list of widget data or singleton
   */  
  usageData(planID):Observable<any>{
    if(!planID){
      this.logger.debug("No planID passed" + planID);
      return Observable.of({});
    }
    return this.db.object(`/usage/${planID}/current`)
      .catch(() => {
	this.logger.warn("No data discovered for planID" + planID);
	return Observable.of({});
      });    
  }

  // Aircraft should be its own service
  acftStatus(acftID):Observable<any>{
    if(!acftID){
      return Observable.of({});
    }
    return this.db.object(`/acstatus/${acftID}/`)
      .catch(() => {
	this.logger.warn("No data discovered for acftID" + acftID);
	return Observable.of({});
      });    

  }

  /**
   * widgetData - Determines the type of widget based upon options
   * @param {Observable<Widget[]>} widgets : set of widgets to get data from
   * @return {Observable<any> } : either a list of widget data or singleton
   */  
  widgetData(widgets):Observable<any>{
    return widgets.map(widget => {
      switch(_.head(_.keys(widget.widget_options))){
      case 'historicalPlanID':
	return this.historicalUsage(165183,{query:{limitToLast:7}});
      case 'planID': // move to usage service maybe
	// logic here for current or historical
	return this.usageData(widget.widget_options.planID);
      case 'planIDs': // move to usage service maybe
	// logic here for current or historical
	return this.multipleUsages(widget.widget_options.planIDs)
      case 'acftID': // move to usage service maybe
	// logic here for current or historical
	return this.acftStatus(widget.widget_options.acftID)

      default:
	this.logger.debug("No options defined for widget");
	return Observable.of({});
      }
    });
  }
  
  /**
   * add - Adds the widget
   * @param {any} widget : widget to add
   * @return {string } : widget types db
   */
  add(widget: any): string{
    return this.db.list(`/dashboard_user/${this.userId}/widgets`).push(widget)
      .then( a => this.logger.debug("Widget added with key " + a))
      .catch( err => this.logger.error("Failed to add Widget"));
  }

  /**
   * remove - Removes the widget
   * @param {string} widget : widget to add
   * @return {any } : widget types db
   */
  remove(key: string){
    return this.widgets.remove(key)
      .then(() => this.logger.debug("widget removed successfully"))
      .catch(err => this.logger.error("Failed to remove Widget"));
  }
  
  /**
   * updateAll - Updates all the widgets;
   * @param {Widgets[]} : widgets to update
   * @return {any } : widget types db
   */
  updateAll(widgets:Widget[]){
    //delete data Node
    let tmpWidgets = widgets;
    widgets.map(widget => { delete widget.data});
    //non-destructive    this.widgets.update('widgets', widgets).then(() => console.log("Widgets updated"));

//destructive    this.widgets.set('widgets',widgets).then(() => console.log("Widgets updated"));
    return tmpWidgets;
  }


  updateOptions(key,options?){
    console.log(key,options);
    // this.db.object(`/dashboard_user/${this.userId}/widgets/${key}/widget_options`)
    //   .set(options)
    //   .then(a => this.logger.debug("Success updating widget options " + a))
    //   .catch(err => this.logger.warn("Failure updating widget options for " + key + "Error " + err));
  }
  /**
   * update - Removes the widget
   * @param {string} widget : widget to add
   * @return {any } : widget types db
   */
  update(key, widget){    
    return this.widgets.update(key, widget)
      .then(() => this.logger.debug("widget updated successfully"))
      .catch(err => this.logger.error("Failure to update widget " + err))
  }
}
