import { Injectable } from '@angular/core';
import {AngularFireDatabase, FirebaseObjectObservable} from 'angularfire2/database';
import { Subscription } from 'rxjs/Subscription';
import { Observable } from 'rxjs/Observable';
import { AuthenticationService } from './authentication.service';
import * as _ from 'lodash';
import { User } from '../model/user';
import { Store } from '@ngrx/store';
import { AppStore } from '../store/app-store';
import { LoginSuccess, Logoff } from '../store/actions/user.actions'
import { CookieService } from 'ngx-cookie';
import { Router } from '@angular/router';


@Injectable()
export class UserService {

  user: User;
  userId = '';
  authSub: Subscription;
  /**
   * Represents UserService
   * @constructor
   */
  constructor(private db: AngularFireDatabase,
	      private authService: AuthenticationService,
	      private store: Store<AppStore>,
	      private cookieService:CookieService,
	      private router:Router
	     ) 
  {
  }

  /**
   * initAuthObs - check the cookie for legacy compatibility
   * @return {Subscription} - subscription for the auth observable
   */


  // I DONT THINK WE NEED THIS SINCE WE HOST THERE ALREADY WE HAVE THE COOKIE Shouldnt server handle this?
  initAuthObs(){ // might be easier to test with if we return Observable to test the sub
    //We need to refactor this 
    return Observable.interval(1000).subscribe(() => {
      // FIX ME BEFORE PROD RELEASE
      // ADD to express server script to return an auth tkt cookie
      const auth_tkt = this.cookieService.get('auth_tkt');
      if (_.isEmpty(auth_tkt)){
	console.log('No Cookie found or no user in local storage');	
        this.logout();
      }
    });
  }
  

  /** logout() - user out of firebase unsubs from auth and clears user **/
  logout(){
    this.cookieService.remove('auth_tkt');
    this.authSub.unsubscribe();
    this.setUser(null, null);
    this.store.dispatch(new Logoff());
    let host = document.location.host;   
    window.location.href = 'https://' + host + '/ADC/Login/Login'    
  }

  /** login() - user out of firebase unsubs from auth and clears user **/
  login(){
    this.authService.authenticate().subscribe(
      fbUser => {
	if (fbUser){
   	  this.user = new User(fbUser);
	  this.authSub = this.initAuthObs();	  
	  fbUser.getIdToken(false).then(toke => { this.user.idToken = toke});
	  this.fetchUser(this.user.userId);
	  this.store.dispatch(new LoginSuccess(this.user))
	}else{
	  this.user = null;
	  // empty local storage and route to the logout
	  this.setUser(null, null);
	}
      }, // route herec
      err => console.log(err));
    return Observable.of(this.user);
  }

 
  /**
   * fetchUser - Subscribes to the firebase user
   * @param userId {string} - userid
   * @return {User} - ts class for user
   */
  fetchUser(userId: string){
    this.getFBUser().subscribe(snap => {
      if (snap){
	this.user.addSnapshot(snap.val());	
	// move this into ngrx store
	this.db.object('/company/' + snap.val().companyID + '/tails')
	  .subscribe(tailsAndPlans => this.user.addPlans(tailsAndPlans));
      }else{
	this.user = null;
      }});
  }

  /**
   * getUser - gets user from firebase
   * @return {FireBaseObjectObservable<any>} the firebase observable representing user
   */
  getFBUser(): FirebaseObjectObservable<any>{
    return this.db.object('/user/' + this.user.userId,
			  {preserveSnapshot: true})
  }
  
  getUser(): Observable<any>{
    if (this.user){
      return  Observable.of(this.user);
    }
  }


  /**
   * getUserId - gets user snapshot
   * @return {any} snapshot of current user
   **/
  getUserSnapshot(){
    return this.user.snap;
  }

   /**
   * setUser - sets the user in localstorage
   * @param user {User} - typescript representation of hte user
   * @param userId {any} - user id
   * @return {any} - the firebase observable representing user
   */
  setUser(user: User, userID: any): void{

    //THIS NEEDS TO BE MOVED TO STORE WE CAN REMOVE THE MODULE
    if (user){
      localStorage.setItem(userID, JSON.stringify(user));
      localStorage.setItem('userID', userID);
    }else{
      localStorage.clear();
    }
  }
}
