import { Component, OnInit, OnDestroy,
	 EventEmitter,
	 Input,
	 Output
       } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { WidgetService } from '../services/widget.service';
import { Router } from '@angular/router';
import { Store } from '@ngrx/store';
import { AddWidget } from '../../core/store/actions/widget.actions';
import { AppStore } from '../../core/store/app-store'
import { WidgetType } from '../model/widget-types'
import * as _ from 'lodash';
import { getUserPlans } from '../../core/store/reducers/user.reducer'


@Component({
  selector: 'app-addw',
  templateUrl: './addw.component.html',
  styleUrls: ['./addw.component.scss'],
})
export class AddwComponent implements OnInit {

  wTypes: WidgetType[];
  isLoading = true;
  selectedWidget: any;
  filteredWidgets  = [];
  form:  FormGroup;
  plans;
  
  
  constructor(
    private store: Store<AppStore>,
    private widgetService:WidgetService,
    private fb: FormBuilder,
    private router: Router
  ) {
    this.store.select(getUserPlans).combineLatest().subscribe(plans => {
      this.plans = plans[0];
      this.createForm();
    });
    
  }

  /** selectWidget - sets the selected widget **/
  selectWidget(widget){
    // The widget is selected
    this.selectWidget = widget;
  }
  
  addWidget(ev){
    this.store.dispatch(new AddWidget(ev.widget));
    this.cancel();
  }
  
  /** cancel - close this dialog or pane or whatever it ends up being **/
  cancel() {
    this.router.navigate(['dashboard']);
  }

  /**
   * filterBy Returns filtered data based on passed params
   * @param filterBy {string} string value to filter on
   **/
  filterBy(filterBy: string){
    // user changes filer and update the main viewport
//    this.filteredWidgets = _.filter(this.wTypes, ['identifier', filterBy]);
  }

  /** createForm - creates the form dropdown boxes **/

  createForm(){   
    let planIds = _.flatten(_.map(
      _.without(_.map(this.plans, 'plan'), undefined, 'None'), _.keys));
    
    this.form = this.fb.group({
      wTypes: this.fb.array(['Plan', 'Tail', 'Application']),
      plans: this.fb.array(planIds),
      tails: this.fb.array(_.map(this.plans, 'tailNumber'))
    });
  }

  /** ngOnInit - get widgets and tail info **/
  ngOnInit() {
    // move to ngrx store
    this.widgetService.types().subscribe(types => {
      this.wTypes = types;
      this.filteredWidgets = types;
      this.isLoading = false;
    });

    this.createForm();
  }

  /** ngOnDestroy - cleanup **/
  ngOnDestroy(){
    this.filteredWidgets = [];
    this.isLoading = true;
    this.form.reset();
  }
}
