import { Component, OnInit, Input, ChangeDetectionStrategy,ViewChild } from '@angular/core';
import { kebabCase, values, map, keys, first, isEmpty } from 'lodash';
import { Widget } from '../../model/widget';
import { Usage } from '../../model/usage';

@Component({
  selector: 'widget-usage',
  templateUrl: './usage.component.html',
  styleUrls: ['./usage.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})

export class UsageComponent implements OnInit {
  // we probably just need to injest the data here but time...
  @Input() widget;
  usageType;
  tailValue;
  show:number;
  textMode = false;
  option: string;
  usage:Usage;
  

  constructor() { }

  /**
   * usageType - sets usage Type for css classes
   * @return {string} - the type of usage for plan
   */  
  setUsageType(){

    switch(true){
    case (this.widget.identifier === 'plan-app-usage'):
      this.usageType = 'APP_USAGE';
      break;
    case (this.widget.identifier === 'tail-usage'):
      this.usageType = 'TAIL_USAGE';
      break;      

    case (this.widget.size.rows === 1 && this.widget.size.cols === 1):
      this.usageType = 'SINGLE_PLAN_1x1' 
      break;
    case (this.widget.size.rows === 1 &&  this.widget.size.cols === 2):
      this.usageType = 'SINGLE_PLAN_1x2'
      break;
    case (this.widget.size.rows === 2 && this.widget.size.cols === 2):      
      this.usageType = 'SINGLE_PLAN_2x2'
      break;
    }
  }
  
  tails(){
    return values(this.widget.data.tails);
  }
  
  onSelect(option: string) {
    this.option = option
    this.tailValue = option
  }
  
  public dataToColumn(){
    if(!this.widget.data) {return [[]]}
    
    switch(this.usageType){
    case 'SINGLE_PLAN_1x1':
      return  [['Usage', this.widget.data.usage],
	       ['Remaining', this.widget.data.planLimit - this.widget.data.usage]];
    case 'SINGLE_PLAN_2x2':
      return map(this.widget.data.tails,
      		 (tail) =>  [tail['aircraftID'], tail['usage']]);      
    case 'APP_USAGE_1x1':
      return map(this.widget.data.tails[this.tailValue].appUsage.slices,
		 (app) =>  [app['app_name'], app['app_usage']]);
      
    default:
      return [[]]    		 
    }
  }
  
  // move the following functions for progress into its own directive
  usagePercent(){
    return (this.widget.data.usage/this.widget.data.planLimit)*100 ;
  }
  setProgressWidth(){
    return this.usagePercent() + '%';
  }
  setProgressLabel(){
    return "used   " + this.widget.data.usage  + " mb ";
    
  }
  progressColor(){
    let diff = this.usagePercent();
    let classStr = " is-large "
    switch(true){
    case diff >= 90:
      return classStr + "is-danger";      
    case diff >= 75:
      return classStr +  "is-warning"; //matches scss variable is-warn
    default:
      return classStr + "is-success";
    }
  }
  progressPercent
  
  epoch(dateString){
    return new Date(dateString).getTime();
  }
  
  /**
   * daysLeft - returns days left in cycle
   *
   * @return {string} - date in days
   **/
  daysLeft(){
    let now = new Date();
    let end = new Date(this.widget.data.endDate);
    let timeDiff = Math.abs(end.getTime() - now.getTime());
    return  Math.ceil(timeDiff / (1000 * 3600 * 24)); 
  }

  toggle(){ }
  
  ngOnInit() {
    if(isEmpty(this.widget.data)){
      this.usageType = 'NO_DATA';
    }else{
      this.widget.data.usage = Math.round(this.widget.data.usage);
      this.usage = this.widget.data.usage;
      this.setUsageType();
    }
  }
}
