import { Component,
	 OnInit,
	 Input,
	 Output,
	 ChangeDetectionStrategy,
	 EventEmitter} from '@angular/core';
import { WidgetService } from '../services/widget.service';
import { WidgetOptions } from '../widget-options';
import { Observable } from 'rxjs/Observable';
import { Widget, SBBWidget } from '../model/widget';
import {times  } from 'lodash'
@Component({
  selector: 'dashboard-widgetx',
  templateUrl: './widgetx.component.html',
  styleUrls: ['./widgetx.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})

export class WidgetxComponent implements OnInit {
  //explain what were doign
  // funneling data to corect places smallest amount of code as possible
  chartType;
  @Input() widget:Widget;
  @Input() editMode:boolean;
  // Improve how this is done 
  @Input() tails?;
  @Input() plans?;
  private widgetIdentifier = '';
  widgetOptions;
  @Output() remove: EventEmitter<string> = new EventEmitter<string>();

  /**
   * Represents Widget
   * @constructor
   */
  constructor() {
  }
  
  /**
   * setChart - sets a chart type
   * @param val {string} - the type of chart to be rendered
   */  
  setChart(val){
    this.chartType = val;
  }
  
  /**
   * removeWidget - removes the widget from firebase
   * @param id {string} - the uniqure id of the widget
   */  
  removeW(){
    this.remove.emit(this.widget.id);
  }
  /**
   * settings - determines settings to be used in app
   * @return widgetOptions {WidgetOptions} - subset of widget data for options
   */    
  setWidgetOptions(){
    return new WidgetOptions(this.widget.identifier,this.widget.data,this.widget.id);
  }
  
  /**
   * resize - auto scales the widget to a different size.
   */  
  resize(){
    this.widget.size = {cols:2,rows:2}
    //event emitter here for parent... Or just subclass gridster item?
  }
  
  ngOnInit() {
    this.widgetOptions = this.setWidgetOptions();
  }
}
