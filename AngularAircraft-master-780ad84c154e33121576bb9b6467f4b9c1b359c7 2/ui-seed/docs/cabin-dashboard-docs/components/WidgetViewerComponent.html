import { Component, OnInit, OnDestroy, Input, Output, EventEmitter} from '@angular/core';
import { WidgetService } from '../../services/widget.service'
import * as _ from 'lodash';


@Component({
  selector: 'app-widget-viewer',
  templateUrl: './widget-viewer.component.html',
  styleUrls: ['./widget-viewer.component.scss']
})
export class WidgetViewerComponent implements OnInit, OnDestroy {

  @Input() filteredWidgets: string[];
  @Output() addingWidget: EventEmitter<any> = new EventEmitter<any>();

  selectedWidget = '';

  constructor() {}

  widgetSelected(widget: string){
    this.selectedWidget = widget;
  }

  ngOnInit() { }

  /**  submit - adds widget to dashboard **/
  addWidget(widget: any) {
    const widgetToAdd = {
      identifier: widget.identifier,
      size : {
	cols:1,
	rows:1
      },
      position: {
	desktop: {
	  x:0,
	  y:0
	}
      }
    };
    this.addingWidget.emit({widget: widgetToAdd});
  }

  ngOnDestroy(){
    this.filteredWidgets = []
    this.selectedWidget = '';
  }
}
