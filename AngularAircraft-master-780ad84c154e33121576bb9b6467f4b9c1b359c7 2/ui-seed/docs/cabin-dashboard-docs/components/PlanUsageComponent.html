import { Component, OnInit, Input, OnDestroy } from '@angular/core';
import { AngularFireDatabase } from 'angularfire2/database';
import { ActivatedRoute, Router } from '@angular/router';

@Component({
  selector: 'plan-usage',
  templateUrl: './plan-usage.component.html',
  styleUrls: ['./plan-usage.component.css']
})
export class PlanUsageComponent implements OnInit {
    @Input('planID')
    planID;
    @Input('isWidget')
    isWidget;
    plan: Object;
    planTailIDs: Array<any>;
    error: string;
    planObj;
    max;
    limit;
    usage;
    options;
    public barChartLabels: string[] = ['SBB Usage'];
    public barChartType = 'horizontalBar';
    public barChartLegend = true;
    chartData;

    constructor(private route: ActivatedRoute, private db: AngularFireDatabase,
                private router: Router) {
        this.plan = undefined;
        this.planTailIDs = [];
        this.error = '';
    }

    private resetChartData(){
        this.options = this.options = {
            'unusedColor': 'rgb(208, 207, 207)',
            'usedColor': 'rgb(57, 253, 86)',
            'exceedColor': 'rgb(228, 102, 102)',
            barThickness: '40px',
            margin: '10px'
        };

    }

    public gotoTailUsage(tailID){
        this.router.navigate(['/tailusage', { planID: this.planID}]);
        // console.log("going to tail usage");
    }

    private getChartData(){
        this.planTailIDs = Object.keys(this.plan['current'].tails);
        const limit = this.plan['current'].planLimit;
        let usage = this.plan['current'].usage;
        const startDate = this.plan['current'].startDate;
        const planPeriod = this.plan['current'].planPeriod;
        let endDate;

        let max = 0;
        // this.colorScheme.domain = [this.greenColor];

        if (usage == 'None'){
            usage = 0;
        }
        max = limit;
        this.options.usedColor = 'rgb(57, 253, 86)';
        if (limit == 'None'){
            max = usage * 1.1 || 100;
        }else if (limit < usage){
            // max = usage * 1.1;
            // this.colorScheme.domain = [this.yellowColor];
        }else if ( usage > (0.85 * limit)){
            this.options.usedColor = 'rgb(255, 214, 0)';
        }

        this.max = +max;
        this.limit = limit;
        this.usage = +usage;

        const data = {
            'limit': this.max,
            'measure': this.usage,
            'measureTooltip': `Current usage: ${usage} MB`,
            'unusedTooltip' : `Remaining usage: ${max - usage} MB`,
            'exceedTooltip' : `Exceeded Limit by ${usage - max} MB`,
        };
        if (this.limit == 'None'){
            data.unusedTooltip = 'No Plan limit.';
        }
        this.chartData = data;


    }

    private getPlanInfo(){

        this.planObj = this.db.object(`/usage/${this.planID}`,
            { preserveSnapshot: true }).subscribe(
                snapshot => {
                    this.error = '';
                    this.planTailIDs = [];
                    this.plan = snapshot.val();
                    if (this.plan && this.plan.hasOwnProperty('current')){
                        this.getChartData();
                    }
                    else{
                        this.error = 'Plan Information is not Available at the moment';
                    }
                }
            );
    }

    ngOnInit() {
        if (! this.planID){
        this.planID = this.route.snapshot.params['planID'];
        }
        this.resetChartData();
        this.getPlanInfo();
    }

    ngOnDestroy(){
        this.planObj.unsubscribe();
    }
}
