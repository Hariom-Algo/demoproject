import { Component, OnInit, OnDestroy, ChangeDetectionStrategy,AfterContentInit } from '@angular/core';
import { GridsterConfig } from 'angular-gridster2/dist/gridsterConfig.interface';
import { Widget } from './model/widget'
import { Store } from '@ngrx/store';
import { RemoveWidget } from '../core/store/actions/widget.actions';
import { AppStore } from '../core/store/app-store'

import * as _ from 'lodash';

@Component({
  selector: 'dashboard',
  templateUrl: './dashboard.component.html',
  styleUrls: ['./dashboard.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class DashboardComponent implements OnInit, OnDestroy {
  // configure these make the dashboard an array of gridster items which will be subclassed
  // maybe something like Widget extends GridsterItem
  options: GridsterConfig;
  dashboard: Array<Object>;
  widgets;
  screenSize: string;
  orphanWidget: boolean;
  // move these to dashboard controls component
  controlsHidden = false;
  editMode = false;

  // type these two
  constructor(private store: Store<AppStore>){
    this.widgets = this.store.select('widgets');
  }
  
  public toggleDragMode(){
    this.editMode = !this.editMode
    this.options.draggable.enabled = !this.options.draggable.enabled;
    this.options.api.optionsChanged();
  }
  
  setOptions() {
    this.options = {
      gridType: 'scrollVertical',
      compactUp: false,
      compactLeft: false,
      mobileBreakpoint: 640,
      itemChangeCallback: this.itemChange.bind(this),
      margin: 6,
      outerMargin: true,
      maxItemCols: 16,
      maxCols: 8,
      minCols: 1,
      minItemCols: 1,
      maxItemRows: 100,
      minItemRows: 1,
      defaultItemCols: 1,
      defaultItemRows: 1,
      fixedColWidth: 150,
      fixedRowHeight: 150,
      draggable: {
        enabled: true,
        stop: this.eventStop.bind(this)
      },
      resizable: {
        enabled: true,
        stop: this.eventStop.bind(this)
      },
      swap: true,
      pushItems: true,
    };

    // this.options = {

  }

  private makeWidgets(){
    // generate a dashboard list with the position, size and metadata required for the widgets
    this.orphanWidget = false;
    this.dashboard = [];
    let nextY = 0;
    if (this.orphanWidget){
      this.toggleDragMode();
    }
  }
  
  
  private save(){
    //this.store.dispatch(new UpdateWidgets(this.widgets));

  }

  toGridster(widget:Widget){        
    return _.merge(widget.position.desktop, widget.size);
  }
  
  
  removeItem(item) {
    this.dashboard.splice(this.dashboard.indexOf(item), 1);
  };

  /**
   *
   * removeWidget - removes the widget
   * @param key - the firebase key for widget
   *
   **/
  removeWidget(key){
    this.store.dispatch(new RemoveWidget({id:key}));
  }
  
  eventStop(item, scope) {
  }
  
  itemChange(item, scope) {
  }
  
  itemInit(item) {
    
  }
  
  ngOnInit() {
    this.dashboard = [];
    // set the login here we might need to pull this out to app component

    this.setOptions();
    this.computeColWidth();
  }
  
  ngAfterContentInit(){
  }
  
  private computeColWidth(){
    const grid: Element = document.querySelector('gridster');
    let width;
    if (grid){
      width = grid['offsetWidth'];
    }else{
      width = window.innerWidth;
    }
    this.options.maxItemCols = 8;
    this.screenSize = 'desktop';
    if (width < 960){
      this.options.maxItemCols = 6;
      this.screenSize = 'laptop';
    }
    if (width < 720){
      this.options.maxItemCols = 4;
      this.screenSize = 'tablet';
    }
    if (width < 450){
             this.options.maxItemCols = 2;
      this.screenSize = 'phone';
    }
    
    const colWidth = Math.floor(width / this.options.maxItemCols);
    this.options.fixedColWidth = colWidth;
    this.options.fixedRowHeight = colWidth;
    this.makeWidgets();
    
  }
  onResize(event){
    this.computeColWidth();
    this.options.api.optionsChanged();
  }
  ngOnDestroy(){
  }
}
