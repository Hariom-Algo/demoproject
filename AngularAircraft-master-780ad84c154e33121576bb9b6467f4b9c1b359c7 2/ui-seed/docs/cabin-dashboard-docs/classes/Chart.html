import { SINGLE_PLAN_1x1,  SINGLE_PLAN_2x2, APP_USAGE_1x1, APP_USAGE_2x2} from '../model/graph'; // need a freaking indexts
import { HISTORICAL_USAGE} from '../model/graph';
import { merge, map, assign,first } from 'lodash';
import * as c3 from  'c3'
import * as d3 from 'd3';

const  widgetGraphMap = {
  'SINGLE_PLAN_1x1': SINGLE_PLAN_1x1,
  'SINGLE_PLAN_2x2': SINGLE_PLAN_2x2,
  'APP_USAGE_1x1': APP_USAGE_1x1,
  'APP_USAGE_2x2': APP_USAGE_2x2,

};

export class Chart{

  chartType = 'pie';
  columnData;
  selector = 'chart';
  widgetGraphType;

  
  // Add chart types take out of component
  // Clean up code later quickly getting out these graphs
  // Think the types should be moved to separate graph models or even Firebase
  constructor(chartType?, data?){
    if(chartType){
      this.chartType = chartType;
    }
  }
  /**  generate quick graph**/
  generate(selector?): c3.ChartAPI{
    if(selector){
      this.selector = selector;
    }
    let binder = {bindto: '.' + this.selector};
    let chart =  c3.generate(merge(binder,APP_USAGE_1x1,{data:{columns:[this.columnData]}}));
    return chart;
  }




  // maybe move to constructor
  generateWidgetGraph(selector, wGraph, columnData): c3.ChartAPI{
    this.selector = selector;
    this.widgetGraphType = wGraph;
    this.columnData = columnData;
    let widGraph = widgetGraphMap[wGraph];
    widGraph.data = merge(widGraph['data'], {columns : this.columnData});
    if(widGraph.data.type === 'donut'){
      widGraph.data = this.mergeMouseOver(widGraph);
      merge(widGraph, { onrendered: () => {
        let ratio = this.donutRatio();
        let color  = '';
	
        if(ratio  >= 90){
          color = '-danger';
        }else if(ratio   >= 75){
          color = '-warn';
        }	
        this.customDonutTitle(this.donutRatio() + '%', columnData[0][0], color);
      }});
    }
    return c3.generate(merge({bindto: '.' + this.selector}, widGraph));
  }


  /*************** Start Donut Related functions - move to its own class *************************** /
  /**  clear the donut title **/
  clearDonutTitle(selector){
    let label = d3.select('div.' + this.selector + ' .c3-chart-arcs-title')
    label.classed(selector, !label.classed(selector));
    d3.select('div.' + this.selector + ' .donut-secondary-title').remove();
    d3.select('div.' + this.selector + ' .c3-chart-arcs-title').text('');
  }
  
  /** create a custom title for the donut chart **/
  customDonutTitle(title, selector, color?)  {
    let top = d3.select('div.' + this.selector + ' .c3-chart-arcs');
    let label = d3.select('div.' + this.selector + ' .c3-chart-arcs-title');
    let secondTitle = d3.select('div.' + this.selector + ' .donut-secondary-title');
    
    if(color){   
      let sel = d3.select('div.' + this.selector + ' g.c3-target-' + selector);
      sel.classed('c3-target-' + selector + color,true)
      label.classed(selector + color, true);
    }
    label.text(title);

    secondTitle.remove();
    top.append('text')
      .text(selector)
      .attr("dy", 25)
      .attr("x", 0)
      .attr('class', 'donut-secondary-title').style("text-anchor", "middle");
  }
  
  /** ratio for donut **/
  donutRatio(){
    let total = this.columnData[1][1] + this.columnData[0][1];
    return Math.round(this.columnData[0][1]/total*100);
  }

  /** set the colors for the donut chart **/
  donutColors(){
    let color = '#4BBF73';
    let ratio = this.donutRatio();
    if(ratio >= 90){
      color = '#d9534f';
    }else if(ratio >= 75){
      color = '#FFDD5D';
    }
    return { colors:{ Usage: color }};
  }

  /** Donut mouse over actions **/
  mergeMouseOver(widGraph){
    // set the color of the text
    let mov = {      
      onmouseover: (d) => {
	let color;
	if(d.ratio * 100  >= 90){
	  color = '-danger';
	}else if(d.ratio * 100  >= 75){
	  color += '-warn';
	}
	this.customDonutTitle(Math.round(d.ratio * 100) + '%', d.name, color);
      },
      onmouseout: (d) => {
	if(d.value === 'Remaining'){
	  this.clearDonutTitle(d.name);
	}
      }
    };  
    return merge(widGraph['data'], mov, this.donutColors());
  }
  
  /*********** END DONUT RELATED FUNCTIONS *********************************/
}
