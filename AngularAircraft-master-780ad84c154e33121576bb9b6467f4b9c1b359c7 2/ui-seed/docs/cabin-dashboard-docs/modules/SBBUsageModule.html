import {
  ModuleWithProviders, NgModule,
  Optional, SkipSelf
} from '@angular/core';

import { CommonModule } from '@angular/common';
import { FlexLayoutModule } from '@angular/flex-layout';
//components
import { SBBUsageComponent } from './sbbusage.component';
import { TailUsageComponent } from './tail-usage/tail-usage.component';
import { PlanAppUsageComponent } from './plan-app-usage/plan-app-usage.component';
import { PlanUsageComponent } from './plan-usage/plan-usage.component';
import { BulletChartComponent } from '../../bullet-chart/bullet-chart.component';

import { SBBUsageService } from './services/sbbusage.service';

@NgModule({
  imports: [
    CommonModule,
    FlexLayoutModule,
  ],
  declarations: [
      BulletChartComponent,
      PlanAppUsageComponent,
      PlanUsageComponent,
      SBBUsageComponent,
      TailUsageComponent,

  ],
  exports: [
      PlanUsageComponent,
      PlanAppUsageComponent,
      TailUsageComponent,
      SBBUsageComponent
  ],
  providers : [
    SBBUsageService
    ]
})

export class SBBUsageModule {
  constructor( @Optional() @SkipSelf() parentModule: SBBUsageModule) {
    if (parentModule) {
      throw new Error(
        'SBBUsageModule is already loaded. Import it in the AppModule only');
    }
  }

  static forRoot(): ModuleWithProviders {
    return {
      ngModule: SBBUsageModule,
      providers: [
      ]
    };
  }
}
