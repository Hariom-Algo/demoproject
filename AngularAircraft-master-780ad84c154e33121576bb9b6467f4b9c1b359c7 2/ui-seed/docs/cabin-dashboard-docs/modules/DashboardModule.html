import {
  ModuleWithProviders, NgModule,
  Optional, SkipSelf
} from '@angular/core';

import { CommonModule } from '@angular/common';
import { FormsModule, ReactiveFormsModule } from '@angular/forms';

// material imports

import { FlexLayoutModule } from '@angular/flex-layout';

// gridster imports
import { GridsterModule } from 'angular-gridster2';

//services
import { UserService } from '../core/services/user.service';
import { WidgetService } from './services/widget.service'

//components
import { DashboardComponent } from './dashboard.component';
import { AddwComponent } from './addw/addw.component';
import { WidgetViewerComponent } from './addw/widget-viewer/widget-viewer.component';
import { RouterModule } from '@angular/router';
import { ChartComponent } from './chart/chart.component';
import { WidgetxComponent } from './widgetx/widgetx.component';
import { LeafletComponent } from './leaflet/leaflet.component';
import { UsageComponent } from './widgetx/usage/usage.component';
import { AcftStatusComponent } from './widgetx/acft-status/acft-status.component';
import { HistoricalComponent } from './widgetx/historical/historical.component';
import { MultiWidgetComponent } from './widgetx/multi-widget/multi-widget.component';
import { TailComponent } from './widgetx/usage/tail/tail.component';
import { AppComponent } from './widgetx/usage/app/app.component';
import { OptionsComponent } from './options/options.component';
//import { StoreModule } from '@ngrx/store';
//import { reducers } from '.reducers';


@NgModule({
  imports: [
    CommonModule,
    FlexLayoutModule,
    GridsterModule,
    FormsModule,
    ReactiveFormsModule,
    RouterModule,
//    StoreModule.forFeature('dashboard', reducers, { })
  ],
  declarations: [
    DashboardComponent,
    WidgetxComponent,
    ChartComponent,
    AddwComponent,
    WidgetViewerComponent,
    LeafletComponent,
    UsageComponent,
    AcftStatusComponent,
    HistoricalComponent,
    MultiWidgetComponent,
    TailComponent,
    AppComponent,
    OptionsComponent
  ],
  exports: [DashboardComponent, AddwComponent],
  providers: [WidgetService ]
})

export class DashboardModule {
  constructor( @Optional() @SkipSelf() parentModule: DashboardModule) {
    if (parentModule) {
      throw new Error(
        'DashboardModule is already loaded. Import it in the AppModule only');
    }
  }

  static forRoot(): ModuleWithProviders {
    return {
      ngModule: DashboardModule,
      providers: [
      ]
    };
  }
}
