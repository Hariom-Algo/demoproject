import {
  ModuleWithProviders, NgModule,
  Optional, SkipSelf, ErrorHandler
} from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormsModule, ReactiveFormsModule } from '@angular/forms';
import { Routes, RouterModule } from '@angular/router';
import { HttpModule } from '@angular/http';
import { AppRoutingModule } from '../routing/app-routing.module'

// environment imports
import { environment } from '../../environments/environment';
import { CookieModule } from 'ngx-cookie';

// components
import { SidenavComponent } from './nav/sidenav/sidenav.component';
import { FramerComponent } from './framer/framer.component';
import { LoginComponent } from './login/login.component';
import { NavComponent } from './nav/nav.component';
import { NavMenuComponent } from './nav/nav-menu/nav-menu.component';
import { SafeResourcePipe } from './shared/pipes/safe-resource.pipe'
import { AcftInterestComponent } from './acft-interest/acft-interest.component';

// services prolly can nix config service soon
import { ConfigurationService  } from './services/configuration.service';
import { AuthenticationService  } from './services/authentication.service';
import { UserService } from './services/user.service';
import { GlobalErrorHandler } from './services/error-handler';

import { ClickOutside } from './click-outside.directive';
import { LoggerModule } from 'ngx-logger';


@NgModule({
  imports: [
    CommonModule,
    HttpModule,
    AppRoutingModule,
    FormsModule,
    ReactiveFormsModule,
    !environment.production ? LoggerModule.forRoot({ level: 'DEBUG'}) : LoggerModule.forRoot({ level: 'WARN'}),
    CookieModule.forRoot()

  ],
  exports: [
    NavComponent,
    SidenavComponent,
    FramerComponent,
    LoginComponent,
    SafeResourcePipe
  ],
  declarations: [
    NavComponent,
    NavMenuComponent,
    SidenavComponent,
    FramerComponent,
    SafeResourcePipe,
    LoginComponent,
    ClickOutside,
    AcftInterestComponent
  ],
  providers: [
    AuthenticationService,
    ConfigurationService,
    UserService,
    {
      provide: ErrorHandler, 
      useClass: GlobalErrorHandler
    }
  ]
})

export class CoreModule {

  constructor( @Optional() @SkipSelf() parentModule: CoreModule) {
    if (parentModule) {
      throw new Error(
        'CoreModule is already loaded. Import it in the AppModule only');
    }
  }

  static forRoot(): ModuleWithProviders {
    return {
      ngModule: CoreModule,
      providers: [
      ]
    };
  }
}
